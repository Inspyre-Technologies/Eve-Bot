def execute(m, s)
  return if check_ignore(m.user)
  questions = s.split("? ")
  answers = [];
  questions.each {|question|
                  question[0] = question[0].upcase
                 answers << "\"#{question.delete("?")}?\" #{shake!}"
                 }
  output = answers.join(". ") + "."
  m.safe_reply output, true
end
