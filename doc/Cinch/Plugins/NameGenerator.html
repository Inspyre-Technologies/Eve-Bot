def execute(m, nameFormat=nil)

    nameFormat = nameFormat.downcase unless nameFormat == nil

    case nameFormat
    when " -vtn"
      nameFormat = "vtn"
    when " -tvn"
        nameFormat = "tvn"
    when nil
        nameFormat = [
            "vtn",
            "tvn"
        ].sample
    end

    verb = File.readlines('docs/band_list_v.txt').sample
    verb = verb.chomp
    noun = File.readlines('docs/band_list_n.txt').sample
    noun = noun.chomp

    if nameFormat == "vtn"
        name = "#{verb} the #{noun}"
        result(m, name)
    end
    if nameFormat == "tvn"
        name = "The #{verb} #{noun}"
        result(m, name)
    end
    return

end
