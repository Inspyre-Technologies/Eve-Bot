def brb(m)
  return if check_ignore(m.user)
  if check_friend(m.user)
    sleep config[:delay] || 3
    m.reply brbr(m)
    return;
  end
  if check_foe(m.user)
    sleep config[:delay] || 3
    m.reply brbr_foe(m)
    return;
  end
  if check_master(m.user)
    sleep config[:delay] || 3
    m.reply brbr(m)
    return;
  end
  sleep config[:delay] || 3
  m.reply brbr(m)
end
